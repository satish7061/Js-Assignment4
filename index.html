    {

    };

Qu 2. Write a program to find the occurance of number 5 digit from given number?

    const Find_Five = (num) => 
    {
       let rem=0,count=0;
       while(num!=0){
          rem=num%10;
          num=parseInt(num/10);
          if (rem==5)
          {
             count++;
          }
        }
        return count;
    };

Qu 3. Write a program to find the sum of all the even integers starting from 1 to N?

    const findSum = (n) => {
        let sum = 0;
        for (let i=0;i<=n;i++){
            if(i%2==0){
              sum += i;
            }
        }
        return sum;
    };

Qu 6. Write a program which ask user for no of lines and print a pattern using an asterik?

    const Print_pattern = (n) => 
    {
      let pattern="";
      for (let i=1;i<=n;i++){
        for (let j=1;j<=i;j++){
           pattern +="*";
        }
        pattern+="\n";
      }
      console.log(pattern);
    };
